-- PROFO Aankoopbeheer - interne meldingen in de app
-- Uitvoeren in de Supabase SQL editor.
-- Doel: bestellingen blijven de officiele bron; meldingen verschijnen in Aankoopbeheer zelf.

begin;

create table if not exists public.aankoop_meldingen (
  id bigint generated by default as identity primary key,
  created_at timestamptz not null default now(),
  gelezen_op timestamptz,
  gebruiker_id bigint not null references public.gebruikers(id) on delete cascade,
  bestelling_id bigint references public.aankoop_bestellingen(id) on delete cascade,
  type text not null default 'algemeen',
  titel text not null,
  boodschap text not null,
  actie_url text not null default '#bestellingen'
);

create index if not exists aankoop_meldingen_gebruiker_idx
  on public.aankoop_meldingen (gebruiker_id, gelezen_op, created_at desc);

create index if not exists aankoop_meldingen_bestelling_idx
  on public.aankoop_meldingen (bestelling_id);

alter table public.aankoop_meldingen enable row level security;

drop policy if exists "aankoop_meldingen_select_eigen_of_beheer" on public.aankoop_meldingen;
drop policy if exists "aankoop_meldingen_update_eigen_of_beheer" on public.aankoop_meldingen;

create policy "aankoop_meldingen_select_eigen_of_beheer"
  on public.aankoop_meldingen
  for select
  to authenticated
  using (
    gebruiker_id = public.current_gebruiker_id()
    or public.current_gebruiker_is_beheerder()
  );

create policy "aankoop_meldingen_update_eigen_of_beheer"
  on public.aankoop_meldingen
  for update
  to authenticated
  using (
    gebruiker_id = public.current_gebruiker_id()
    or public.current_gebruiker_is_beheerder()
  )
  with check (
    gebruiker_id = public.current_gebruiker_id()
    or public.current_gebruiker_is_beheerder()
  );

create or replace function public.aankoop_melding_toevoegen(
  p_gebruiker_id bigint,
  p_bestelling_id bigint,
  p_type text,
  p_titel text,
  p_boodschap text,
  p_actie_url text default '#bestellingen'
)
returns bigint
language plpgsql
security definer
set search_path = public
as $$
declare
  v_huidige_gebruiker_id bigint;
  v_melding_id bigint;
begin
  v_huidige_gebruiker_id := public.current_gebruiker_id();

  if v_huidige_gebruiker_id is null then
    raise exception 'Geen actieve PROFO-gebruiker gevonden voor deze melding.';
  end if;

  if not exists (
    select 1
    from public.gebruikers gebruiker
    where gebruiker.id = p_gebruiker_id
      and coalesce(gebruiker.actief, true) = true
  ) then
    return null;
  end if;

  insert into public.aankoop_meldingen (
    gebruiker_id,
    bestelling_id,
    type,
    titel,
    boodschap,
    actie_url
  )
  values (
    p_gebruiker_id,
    p_bestelling_id,
    left(coalesce(nullif(trim(p_type), ''), 'algemeen'), 80),
    left(coalesce(nullif(trim(p_titel), ''), 'Melding Aankoopbeheer'), 180),
    left(coalesce(nullif(trim(p_boodschap), ''), 'Er is een wijziging in Aankoopbeheer.'), 1000),
    coalesce(nullif(trim(p_actie_url), ''), '#bestellingen')
  )
  returning id into v_melding_id;

  return v_melding_id;
end;
$$;

grant execute on function public.aankoop_melding_toevoegen(bigint, bigint, text, text, text, text)
  to authenticated;

commit;

select
  'Interne meldingen voor PROFO Aankoopbeheer zijn actief.' as resultaat,
  count(*) as bestaande_meldingen
from public.aankoop_meldingen;
