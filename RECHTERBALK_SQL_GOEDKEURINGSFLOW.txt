-- PROFO Aankoopbeheer - goedkeuringsflow en rollen
-- Uit te voeren in Supabase SQL Editor.
-- Doel:
-- 1. rollen voor aankoopbeheer scherp zetten;
-- 2. Timothy organisatiebreed laten goedkeuren;
-- 3. regiodirecteurs technisch klaarzetten als goedkeurders;
-- 4. bestellingen alleen vrijgeven voor externe invoer na goedkeuring.
--
-- Belangrijk:
-- De exacte teamkoppelingen per regiodirecteur moeten nog uit de SharePoint-teamlijst
-- worden overgenomen in het invulblok hieronder. Codex kon de SharePoint-link niet
-- rechtstreeks lezen omdat die achter Microsoft-authenticatie zit.

begin;

update public.gebruikers
set rol = 'Gebruiker'
where rol is null or btrim(rol) = '';

update public.gebruikers
set rol = 'Superadmin',
    functie = coalesce(nullif(functie, ''), 'Aankoopbeheer')
where lower(email) = 'jorn.neeus@profo.be'
   or lower(naam) = 'jorn neeus';

update public.gebruikers
set rol = 'Beheerder aankoop',
    functie = coalesce(nullif(functie, ''), 'Aankoopbeheer')
where lower(email) = 'kathleen.nerinckx@profo.be'
   or lower(naam) = 'kathleen nerinckx';

update public.gebruikers
set rol = 'Algemeen directeur',
    functie = coalesce(nullif(functie, ''), 'Algemeen directeur')
where lower(naam) in ('timothy van raemdonck', 'timothy vanraemdonck')
   or lower(email) like 'timothy.%@profo.be';

update public.gebruikers
set rol = 'Regiodirecteur',
    functie = coalesce(nullif(functie, ''), 'Regiodirecteur')
where lower(naam) in (
  'karima lakdim',
  'nathan blondeel',
  'annelies vuye',
  'joke delille',
  'joke de lille'
);

create table if not exists public.aankoop_goedkeurder_scopes (
  id bigint generated by default as identity primary key,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  goedkeurder_id bigint not null references public.gebruikers(id) on delete cascade,
  scope_type text not null default 'teamlid',
  teamlid_id bigint references public.gebruikers(id) on delete cascade,
  locatie_id bigint references public.locaties(id) on delete cascade,
  actief boolean not null default true,
  constraint aankoop_goedkeurder_scopes_scope_type_check
    check (scope_type in ('organisatie', 'teamlid', 'locatie')),
  constraint aankoop_goedkeurder_scopes_scope_check
    check (
      (scope_type = 'organisatie' and teamlid_id is null and locatie_id is null)
      or (scope_type = 'teamlid' and teamlid_id is not null and locatie_id is null)
      or (scope_type = 'locatie' and locatie_id is not null and teamlid_id is null)
    )
);

create unique index if not exists aankoop_goedkeurder_scopes_unique_idx
  on public.aankoop_goedkeurder_scopes (
    goedkeurder_id,
    scope_type,
    coalesce(teamlid_id, -1),
    coalesce(locatie_id, -1)
  );

alter table public.aankoop_goedkeurder_scopes enable row level security;

create or replace function public.current_gebruiker_is_goedkeurder()
returns boolean
language sql
stable
security definer
set search_path = public
as $$
  select coalesce(public.current_gebruiker_rol(), '') in (
    'Goedkeurder aankoop',
    'Algemeen directeur',
    'Regiodirecteur',
    'Beheerder aankoop',
    'Superadmin'
  );
$$;

create or replace function public.current_gebruiker_mag_bestelling_goedkeuren(
  p_bestelling public.aankoop_bestellingen
)
returns boolean
language sql
stable
security definer
set search_path = public
as $$
  select
    public.current_gebruiker_is_beheerder()
    or exists (
      select 1
      from public.aankoop_goedkeurder_scopes scope
      where scope.actief = true
        and scope.goedkeurder_id = public.current_gebruiker_id()
        and (
          scope.scope_type = 'organisatie'
          or (scope.scope_type = 'teamlid' and scope.teamlid_id in (p_bestelling.besteller_id, p_bestelling.aangemaakt_door_id))
          or (scope.scope_type = 'locatie' and scope.locatie_id = p_bestelling.locatie_id)
        )
    );
$$;

revoke all on function public.current_gebruiker_is_goedkeurder() from public;
revoke all on function public.current_gebruiker_is_goedkeurder() from anon;
revoke all on function public.current_gebruiker_mag_bestelling_goedkeuren(public.aankoop_bestellingen) from public;
revoke all on function public.current_gebruiker_mag_bestelling_goedkeuren(public.aankoop_bestellingen) from anon;
grant execute on function public.current_gebruiker_is_goedkeurder() to authenticated;
grant execute on function public.current_gebruiker_mag_bestelling_goedkeuren(public.aankoop_bestellingen) to authenticated;

drop policy if exists aankoop_goedkeurder_scopes_select on public.aankoop_goedkeurder_scopes;
create policy aankoop_goedkeurder_scopes_select
on public.aankoop_goedkeurder_scopes
for select
to authenticated
using (
  public.current_gebruiker_is_beheerder()
  or goedkeurder_id = public.current_gebruiker_id()
);

drop policy if exists aankoop_goedkeurder_scopes_beheerder on public.aankoop_goedkeurder_scopes;
create policy aankoop_goedkeurder_scopes_beheerder
on public.aankoop_goedkeurder_scopes
for all
to authenticated
using (public.current_gebruiker_is_beheerder())
with check (public.current_gebruiker_is_beheerder());

-- Timothy krijgt organisatiebreed zicht en goedkeuringsrecht.
insert into public.aankoop_goedkeurder_scopes (goedkeurder_id, scope_type, actief)
select gebruiker.id, 'organisatie', true
from public.gebruikers gebruiker
where gebruiker.actief = true
  and (
    lower(gebruiker.naam) in ('timothy van raemdonck', 'timothy vanraemdonck')
    or lower(gebruiker.email) like 'timothy.%@profo.be'
  )
on conflict do nothing;

-- Invulblok teamkoppelingen.
-- Vul hier de exacte namen aan vanuit de SharePoint-teamlijst.
-- Laat dit blok leeg zolang de teamlijst nog niet is overgenomen.
create temporary table profo_aankoop_teamkoppelingen (
  goedkeurder_naam text not null,
  teamlid_naam text not null
) on commit drop;

-- Voorbeeld, pas zelf aan met echte rijen uit de teamlijst:
-- insert into profo_aankoop_teamkoppelingen (goedkeurder_naam, teamlid_naam) values
--   ('Karima Lakdim', 'Naam Teamlid 1'),
--   ('Nathan Blondeel', 'Naam Teamlid 2'),
--   ('Annelies Vuye', 'Naam Teamlid 3'),
--   ('Joke DeLille', 'Naam Teamlid 4');

insert into public.aankoop_goedkeurder_scopes (goedkeurder_id, scope_type, teamlid_id, actief)
select goedkeurder.id, 'teamlid', teamlid.id, true
from profo_aankoop_teamkoppelingen koppeling
join public.gebruikers goedkeurder
  on lower(goedkeurder.naam) = lower(koppeling.goedkeurder_naam)
join public.gebruikers teamlid
  on lower(teamlid.naam) = lower(koppeling.teamlid_naam)
where goedkeurder.actief = true
  and teamlid.actief = true
on conflict do nothing;

drop policy if exists aankoop_bestellingen_select_betrokken_of_beheerder on public.aankoop_bestellingen;
create policy aankoop_bestellingen_select_betrokken_of_beheerder
on public.aankoop_bestellingen
for select
to authenticated
using (
  public.current_gebruiker_is_beheerder()
  or besteller_id = public.current_gebruiker_id()
  or aangemaakt_door_id = public.current_gebruiker_id()
  or public.current_gebruiker_mag_bestelling_goedkeuren(aankoop_bestellingen)
);

drop policy if exists aankoop_bestellingen_update_beheerder on public.aankoop_bestellingen;
drop policy if exists aankoop_bestellingen_update_beheerder_of_goedkeurder on public.aankoop_bestellingen;
create policy aankoop_bestellingen_update_beheerder_of_goedkeurder
on public.aankoop_bestellingen
for update
to authenticated
using (
  public.current_gebruiker_is_beheerder()
  or (
    public.current_gebruiker_mag_bestelling_goedkeuren(aankoop_bestellingen)
    and coalesce(status, '') in ('Ter goedkeuring', 'Ingediend', 'Nieuw', 'Extra informatie gevraagd')
  )
)
with check (
  public.current_gebruiker_is_beheerder()
  or (
    public.current_gebruiker_mag_bestelling_goedkeuren(aankoop_bestellingen)
    and coalesce(status, '') in ('Goedgekeurd', 'Geweigerd', 'Extra informatie gevraagd')
  )
);

drop policy if exists aankoop_bestelregels_select_betrokken_of_beheerder on public.aankoop_bestelregels;
create policy aankoop_bestelregels_select_betrokken_of_beheerder
on public.aankoop_bestelregels
for select
to authenticated
using (
  exists (
    select 1
    from public.aankoop_bestellingen bestelling
    where bestelling.id = aankoop_bestelregels.bestelling_id
      and (
        public.current_gebruiker_is_beheerder()
        or bestelling.besteller_id = public.current_gebruiker_id()
        or bestelling.aangemaakt_door_id = public.current_gebruiker_id()
        or public.current_gebruiker_mag_bestelling_goedkeuren(bestelling)
      )
  )
);

commit;

select
  'rollen goedkeurders' as controle,
  naam,
  email,
  rol,
  actief
from public.gebruikers
where lower(naam) in (
  'jorn neeus',
  'kathleen nerinckx',
  'timothy van raemdonck',
  'timothy vanraemdonck',
  'karima lakdim',
  'nathan blondeel',
  'annelies vuye',
  'joke delille',
  'joke de lille'
)
order by rol, naam;

select
  'goedkeuringsscopes' as controle,
  goedkeurder.naam as goedkeurder,
  scope.scope_type,
  teamlid.naam as teamlid,
  locatie.naam as locatie,
  scope.actief
from public.aankoop_goedkeurder_scopes scope
join public.gebruikers goedkeurder on goedkeurder.id = scope.goedkeurder_id
left join public.gebruikers teamlid on teamlid.id = scope.teamlid_id
left join public.locaties locatie on locatie.id = scope.locatie_id
order by goedkeurder.naam, scope.scope_type, teamlid.naam, locatie.naam;
