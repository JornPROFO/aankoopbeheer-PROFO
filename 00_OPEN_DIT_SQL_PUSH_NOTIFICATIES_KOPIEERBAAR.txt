-- PROFO Aankoopbeheer - pushnotificaties
-- Uitvoeren in de Supabase SQL editor.
-- Doel: per toestel een push-abonnement bewaren, met RLS en beperkte toegang.

begin;

create table if not exists public.aankoop_push_abonnementen (
  id bigint generated by default as identity primary key,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  laatste_fout_op timestamptz,
  laatste_fout text,
  actief boolean not null default true,
  gebruiker_id bigint not null references public.gebruikers(id) on delete cascade,
  endpoint text not null unique,
  p256dh text not null,
  auth text not null,
  user_agent text,
  platform text
);

create index if not exists aankoop_push_abonnementen_gebruiker_idx
  on public.aankoop_push_abonnementen (gebruiker_id, actief, updated_at desc);

alter table public.aankoop_push_abonnementen enable row level security;

drop policy if exists "aankoop_push_select_eigen_of_beheer" on public.aankoop_push_abonnementen;
drop policy if exists "aankoop_push_insert_eigen" on public.aankoop_push_abonnementen;
drop policy if exists "aankoop_push_update_eigen_of_beheer" on public.aankoop_push_abonnementen;
drop policy if exists "aankoop_push_delete_eigen" on public.aankoop_push_abonnementen;

create policy "aankoop_push_select_eigen_of_beheer"
  on public.aankoop_push_abonnementen
  for select
  to authenticated
  using (
    gebruiker_id = public.current_gebruiker_id()
    or public.current_gebruiker_is_beheerder()
  );

create policy "aankoop_push_insert_eigen"
  on public.aankoop_push_abonnementen
  for insert
  to authenticated
  with check (
    gebruiker_id = public.current_gebruiker_id()
  );

create policy "aankoop_push_update_eigen_of_beheer"
  on public.aankoop_push_abonnementen
  for update
  to authenticated
  using (
    gebruiker_id = public.current_gebruiker_id()
    or public.current_gebruiker_is_beheerder()
  )
  with check (
    gebruiker_id = public.current_gebruiker_id()
    or public.current_gebruiker_is_beheerder()
  );

create policy "aankoop_push_delete_eigen"
  on public.aankoop_push_abonnementen
  for delete
  to authenticated
  using (
    gebruiker_id = public.current_gebruiker_id()
  );

create or replace function public.aankoop_push_abonnement_opslaan(
  p_endpoint text,
  p_p256dh text,
  p_auth text,
  p_user_agent text default null,
  p_platform text default null
)
returns bigint
language plpgsql
security definer
set search_path = public
as $$
declare
  v_gebruiker_id bigint;
  v_abonnement_id bigint;
begin
  v_gebruiker_id := public.current_gebruiker_id();

  if v_gebruiker_id is null then
    raise exception 'Geen actieve PROFO-gebruiker gevonden voor pushmeldingen.';
  end if;

  if nullif(trim(p_endpoint), '') is null
    or nullif(trim(p_p256dh), '') is null
    or nullif(trim(p_auth), '') is null then
    raise exception 'Pushabonnement is onvolledig.';
  end if;

  insert into public.aankoop_push_abonnementen (
    gebruiker_id,
    endpoint,
    p256dh,
    auth,
    user_agent,
    platform,
    actief,
    updated_at,
    laatste_fout_op,
    laatste_fout
  )
  values (
    v_gebruiker_id,
    p_endpoint,
    p_p256dh,
    p_auth,
    p_user_agent,
    p_platform,
    true,
    now(),
    null,
    null
  )
  on conflict (endpoint) do update
    set gebruiker_id = excluded.gebruiker_id,
        p256dh = excluded.p256dh,
        auth = excluded.auth,
        user_agent = excluded.user_agent,
        platform = excluded.platform,
        actief = true,
        updated_at = now(),
        laatste_fout_op = null,
        laatste_fout = null
  returning id into v_abonnement_id;

  return v_abonnement_id;
end;
$$;

create or replace function public.aankoop_push_abonnement_deactiveren(
  p_endpoint text
)
returns void
language plpgsql
security definer
set search_path = public
as $$
declare
  v_gebruiker_id bigint;
begin
  v_gebruiker_id := public.current_gebruiker_id();

  if v_gebruiker_id is null then
    return;
  end if;

  update public.aankoop_push_abonnementen
  set actief = false,
      updated_at = now()
  where endpoint = p_endpoint
    and gebruiker_id = v_gebruiker_id;
end;
$$;

grant execute on function public.aankoop_push_abonnement_opslaan(text, text, text, text, text)
  to authenticated;

grant execute on function public.aankoop_push_abonnement_deactiveren(text)
  to authenticated;

commit;

select
  'Pushabonnementen voor PROFO Aankoopbeheer zijn voorbereid.' as resultaat,
  count(*) as bestaande_abonnementen
from public.aankoop_push_abonnementen;
